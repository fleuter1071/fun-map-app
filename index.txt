  function setProgress(done, total, failed = 0){
    if(!progressTrackEl || !progressFillEl) return;
    const pct = total ? Math.max(0, Math.min(1, done / total)) : 0;
    progressTrackEl.style.display = "inline-flex";
    progressFillEl.style.width = `${(pct * 100).toFixed(1)}%`;

  function setStatus(msg){ if(statusEl) statusEl.textContent = msg; }

  function loadScript(url){
    return new Promise((resolve, reject) => {
      const s = document.createElement('script');
      s.src = url;
      s.async = true;
      s.onload = () => resolve(url);
      s.onerror = () => reject(new Error('Failed to load ' + url));
      document.head.appendChild(s);
    });
  }

  async function loadWithFallback(urls, name){
    let lastErr;
    for(const u of urls){
      try { await loadScript(u); return u; }
      catch(e){ lastErr = e; }
    }
    throw new Error(`Could not load ${name}. Last error: ${lastErr && lastErr.message}`);
  }

  async function start(){
    try {
      setStatus('Loading libraries…');
      await loadWithFallback([
        'https://cdn.jsdelivr.net/npm/d3@7/dist/d3.min.js',
        'https://unpkg.com/d3@7/dist/d3.min.js',
        'https://cdnjs.cloudflare.com/ajax/libs/d3/7.9.0/d3.min.js'
      ], 'd3');
      await loadWithFallback([
        'https://cdn.jsdelivr.net/npm/topojson-client@3/dist/topojson-client.min.js',
        'https://unpkg.com/topojson-client@3/dist/topojson-client.min.js',
        'https://cdnjs.cloudflare.com/ajax/libs/topojson-client/3.1.0/topojson-client.min.js'
      ], 'topojson-client');

      if(!window.d3 || !window.topojson){
        throw new Error('Libraries loaded but not available on window (d3/topojson).');
      }
      startApp();

    } catch (e) {
      console.error(e);
      setStatus('Failed to load required libraries. Your network may be blocking CDNs. Open DevTools → Console for details.');
    }
  }

  async function fetchJsonWithTimeout(url, timeoutMs=12000){
    const controller = new AbortController();
    const t = setTimeout(() => controller.abort(), timeoutMs);
    try {
      const res = await fetch(url, {signal: controller.signal, headers: {'Accept':'application/json'}});
      if(!res.ok) throw new Error(`HTTP ${res.status} for ${url}`);
      return await res.json();
    } finally {
      clearTimeout(t);
    }

  async function ensurePlaceIndex(stateFips){
    const key = censusPlacesKey(stateFips);
    const cached = readLS(key);

    const url = `${CENSUS_BASE}?get=NAME&for=place:*&in=state:${stateFips}`;

    const entries = [];
    for(let i=1;i<rows.length;i++){
      const r = rows[i];
      const k = normPlaceLabel(String(r[0]).split(",")[0]);
      if(k && r[2]) entries.push([k, r[2], r[0]]);
    }
    writeLS(key, {ts: Date.now(), entries});

  }

  function pickBestPlace(entries, cityName){
    const cityKey = normBasic(cityName);
    if(!cityKey) return null;
    let exact = entries.find(e => e[0] === cityKey);
    if(exact) return exact;
    const candidates = entries.filter(e => e[0].startsWith(cityKey) || cityKey.startsWith(e[0]) || e[0].includes(cityKey));
    if(candidates.length === 0) return null;
    candidates.sort((a,b) => {
      const aSW = a[0].startsWith(cityKey) ? 0 : 1; const bSW = b[0].startsWith(cityKey) ? 0 : 1;
      if(aSW !== bSW) return aSW - bSW;
      const da = Math.abs(a[0].length - cityKey.length); const db = Math.abs(b[0].length - cityKey.length);
      if(da !== db) return da - db;
      return a[0].length - b[0].length;
    });
    return candidates[0];
  }

  function format3DayStrip(city){
    const wx = city?._wx;
    const parts = [0,1,2].map(i => {
      const h = (wx?.hi?.[i] != null) ? Math.round(wx.hi[i]) : "—";
      const l = (wx?.lo?.[i] != null) ? Math.round(wx.lo[i]) : "—";
      const p = (wx?.precip?.[i] != null) ? ` · 💧${Math.round(wx.precip[i])}%` : "";
      const txt = `${formatDOW(wx?.dates?.[i])}: H ${h}° / L ${l}°${p}`;
      
      const currentDayIdx = Math.floor(selectedHourIndex / 24);
      return (i === currentDayIdx) ? `<span class="sel">${txt}</span>` : txt;
    });
    return parts.join("  |  ");
  }

  function getNext24Temps(wx){
    const times = wx?.hourly?.time ?? [];
    const temps = wx?.hourly?.temp ?? [];
    if(!times.length || !temps.length) return null;
    let idx = 0, ct = wx?.current?.time;
    if(ct){
      idx = times.indexOf(ct);
      if(idx < 0 && ct.length >= 13) { const h = ct.slice(0,13)+":00"; idx = times.indexOf(h); if(idx < 0) idx = times.findIndex(t => t >= h); }
      if(idx < 0) idx = 0;
    }
    const slice = temps.slice(idx, idx + 24).filter(v => v != null && isFinite(v));
    return slice.length >= 2 ? slice : null;
  }

  function sparklineSVG(temps){
    const W = 230, H = 46, P = 4, n = temps.length;
    let min = Infinity, max = -Infinity;
    for (const t of temps){ if(t < min) min = t; if(t > max) max = t; }
    if(!isFinite(min) || !isFinite(max)) return "";
    const span = (max - min) || 1;
    const x = (i) => P + (i * (W - 2*P) / (n - 1));
    const y = (t) => P + ((max - t) * (H - 2*P) / span);
    let d = ""; for(let i=0;i<n;i++) d += (i===0 ? "M" : "L") + x(i).toFixed(2) + "," + y(temps[i]).toFixed(2);
    const dArea = d + `L${x(n-1).toFixed(2)},${(H-P).toFixed(2)}L${x(0).toFixed(2)},${(H-P).toFixed(2)}Z`;
    return `<div class="sparkWrap"><div class="sparkHead"><div class="sparkTitle">Next 24h temp</div><div class="sparkMinMax">${Math.round(min)}°–${Math.round(max)}°</div></div><div class="sparkBox"><svg viewBox="0 0 ${W} ${H}" preserveAspectRatio="none"><path class="sparkArea" d="${dArea}"></path><path class="sparkLine" d="${d}"></path><circle class="sparkEnd" cx="${x(n-1).toFixed(2)}" cy="${y(temps[n-1]).toFixed(2)}" r="2.6"></circle></svg></div></div>`;
  }
  function sparklineBlockForCity(city){ const temps = getNext24Temps(city?._wx); return temps ? sparklineSVG(temps) : ""; }

  function debounce(fn, ms = 120) { let t; return (...args) => { clearTimeout(t); t = setTimeout(() => fn(...args), ms); }; }

  async function asyncPool(limit, items, iteratorFn) {
    const ret = [], executing = [];
    for (const item of items) {
      const p = Promise.resolve().then(() => iteratorFn(item)); ret.push(p);
      if (limit <= items.length) {
        const e = p.then(() => executing.splice(executing.indexOf(e), 1));
        executing.push(e); if (executing.length >= limit) await Promise.race(executing);
      }
    }
    return Promise.allSettled(ret);
  }

  let colorScale = null, lastMinHigh = null, lastMaxHigh = null;

  function updateSurfaceOverlay(animated){
    if(!surfaceEnabled || !mapWidth || !mapHeight || !colorScale){ gSurface.style("display","none"); return; }
    const pts = [];
    for(const c of CITIES){
      const v = getSelectedMetric(c);
      if(v != null && c._xy) pts.push({x: c._xy[0], y: c._xy[1], v});
    }
    if(pts.length < 3){ gSurface.style("display","none"); return; }
    gSurface.style("display", null);

    const cell = SURFACE_CELL_PX;
    const gw = Math.max(30, Math.round(mapWidth / cell));
    const gh = Math.max(20, Math.round(mapHeight / cell));
    const values = new Array(gw * gh);
    for(let j=0; j<gh; j++){
      const y = j * cell;
      for(let i=0; i<gw; i++){
        const x = i * cell;
        let num = 0, den = 0;
        for(const p of pts){
          const d2 = (x - p.x)**2 + (y - p.y)**2;
          if(d2 > 270400) continue; // 520^2 max radius
          const w = 1 / (d2 + 1e-6); // power = 2
          num += w * p.v; den += w;
        }
        values[j*gw + i] = (den > 0) ? (num/den) : NaN;
      }
    }

    const domain = colorScale.domain();
    const thresholds = d3.ticks(domain[0], domain[1], 14);
    const contours = d3.contours().size([gw, gh]).thresholds(thresholds)(values);
    const contourPath = d3.geoPath(d3.geoIdentity().scale(cell));

    gSurface.selectAll("path.contour").data(contours, d => d.value).join(
      enter => enter.append("path").attr("class", "contour").attr("d", contourPath).attr("fill", d => colorScale(d.value)).attr("opacity", 0.0).call(enter => enter.transition().duration(animated ? 320 : 0).attr("opacity", 0.38)),
      update => update.call(update => animated ? update.transition().duration(320).attr("d", contourPath).attr("fill", d => colorScale(d.value)) : update.attr("d", contourPath).attr("fill", d => colorScale(d.value))),
      exit => exit.call(exit => exit.transition().duration(200).attr("opacity", 0).remove())
    );
  }

  function computeScaleFromLoaded(animated = true) {
    const vals = [];
    for (const c of CITIES) { const v = getSelectedMetric(c); if (typeof v === "number" && isFinite(v)) vals.push(v); }
    if (vals.length < 2) { applyDotColors(false); updateLegend(); return; }
    
    let vMin = d3.min(vals), vMax = d3.max(vals);
    if (vMin === vMax) { vMin -= 1; vMax += 1; }
    lastMinHigh = vMin; lastMaxHigh = vMax;

    if (colorMode === "precip") {
      colorScale = d3.scaleSequential().domain([Math.max(0, vMin) - 2, Math.min(100, vMax) + 2]).interpolator(d3.interpolateBlues).clamp(true);
    } else {
      const pad = Math.max(1, (vMax - vMin) * 0.06);
      colorScale = d3.scaleSequential().domain([vMin - pad, vMax + pad]).interpolator(d3.interpolateTurbo).clamp(true);
    }
    applyDotColors(!!animated);
    updateLegend();
  }

  async function loadAllWeather(opts = {}) {
    const force = !!opts.force;
    let done = 0, failed = 0, cacheHits = 0;

    for (const c of CITIES) {
      c._wxError = false; c._pulsed = false;
      const cached = readCache(c);
      if (cached) { c._wx = cached.payload; c._wxMeta = { source: "cache", fetchedAt: cached.fetchedAt }; cacheHits++; }
      else { c._wx = undefined; c._wxMeta = { source: "none", fetchedAt: null }; }
    }
    computeScaleFromLoaded(false); applyDotColors(false); if (pinned.size > 0) renderPinnedPanel();
    setProgress(0, CITIES.length, 0);
    setStatus(cacheHits > 0 ? `Loaded cached weather for ${cacheHits}/${CITIES.length} cities. Refreshing…` : `Fetching weather… (0/${CITIES.length})`);

    await asyncPool(CONCURRENCY, CITIES, async (city) => {
      if (force || shouldRefreshFromNetwork(city)) {
        try {
          const wx = await fetchWeatherNetwork(city);
          city._wx = wx; city._wxError = false; city._wxMeta = { source: "live", fetchedAt: Date.now() };
          writeCache(city, wx, city._wxMeta.fetchedAt);
          if (!city._pulsed) { city._pulsed = true; pulseDotOnce(city); }
          refreshTooltipIfHovering(city);
        } catch (err) {
          if (!(city._wx && city._wxMeta?.source === "cache")) { city._wx = null; city._wxError = true; failed++; }
        }
      }
      computeScaleFromLoaded(true); if (pinned.size > 0) renderPinnedPanel();
      done++; setProgress(done, CITIES.length, failed); setStatus(`Fetching weather… (${done}/${CITIES.length})`);
    });

    setStatus(`Done. Updated ${CITIES.length - failed}/${CITIES.length} cities${failed ? ` (${failed} unavailable)` : ""}.`);
    computeScaleFromLoaded(false); applyDotColors(false);
    requestAnimationFrame(() => { computeScaleFromLoaded(false); applyDotColors(false); });
    renderPinnedPanel();
    
    // Automatically find the "current" hour once data loads
    if (selectedHourIndex === 0) {
      const currentHrStr = new Date().toISOString().slice(0,13) + ":00";
      const sample = CITIES.find(c => c._wx?.hourly?.time);
      if (sample) {
        const idx = sample._wx.hourly.time.findIndex(t => t >= currentHrStr);
        if (idx !== -1) {
          selectedHourIndex = idx;
          if (daySlider) daySlider.value = selectedHourIndex;
          updateDayLabelUI();
          computeScaleFromLoaded(false);
        }
      }
    }
  }
